%% macros

%% Images
\newcommand{\includeimagebase}[2][keepaspectratio=true]{%
	\includegraphics[#1]{img/#2}
}
\newcommand{\includeimage}[2][scale=1.0]{
	\begin{center}
	\includeimagebase[#1,keepaspectratio=true]{#2}
	\end{center}
}
\long\def\includeimagefigureM@[#1][#2][#3]#4#5{
	\begin{figure}[#2]
	\centering
	\vspace{-0.3em}
	\includeimagebase[#1,keepaspectratio=true]{#4}
	\vspace{-1em}
	\ifthenelse{\equal{#3}{}}{
		\caption{#5}
	}{
		\caption[#3]{#5}
	}
	\label{img:#4}
	\vspace{-0.5em}
	\end{figure}
}
\newcommand{\includeimagefigure}{\optparams{\includeimagefigureM@}{[scale=1.0][H][]}}

\newcommand{\textref}[2]{\hyperref[#1]{#2 [\shortref{#1}]}}
